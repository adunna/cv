\resheading{Research}

\begin{itemize}

\item[]

	\ressubheading{Calipr Lab}{Amherst, MA}{Advisor: Phillipa Gill}{Jan. 2017 -- Current}

  \begin{itemize}
		\resitem{\textbf{Multi-CDN}\newline A study into performances of CDNs over time, varied by country, source AS, destination AS, and client. Compared local vs. remote caching, and do studies on developing regions and IPV4 vs. IPV6. Pinpointed strategies in Microsoft's and Apple's deployment of CDNs for delivering software updates to clients, and identified impacts of client CDN migration and changes in CDN routing.\newline}

    \resitem{\textbf{China's Firewall and Unpublished Tor Bridges}\newline A revisit to a series of papers published in 2012 and 2015, taking an updated look at how the Great Firewall of China (GFC) blocks unpublished tor relays, specifically bridge relays. We performed in-depth fingerprinting of GFC active scanners, how the GFC performs deep packet inspection (DPI) to determine the presence of Tor traffic, and proposed and tested circumvention methods for Chinese Tor users.}

	\end{itemize}

\end{itemize}
